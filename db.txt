
CREATE DATABASE advanced_train_db;
USE advanced_train_db;

-- 2. USERS (Handles both passengers and admins)
CREATE TABLE Users (
    user_id INT AUTO_INCREMENT PRIMARY KEY,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    phone VARCHAR(15) UNIQUE NOT NULL,
    role ENUM('ADMIN', 'USER') DEFAULT 'USER',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 3. STATIONS (All available railway stations)
CREATE TABLE Stations (
    station_id INT AUTO_INCREMENT PRIMARY KEY,
    station_code VARCHAR(10) UNIQUE NOT NULL,
    station_name VARCHAR(100) NOT NULL,
    city VARCHAR(50),
    state VARCHAR(50)
);

-- 4. TRAINS (Master list of trains)
CREATE TABLE Trains (
    train_id INT AUTO_INCREMENT PRIMARY KEY,
    train_number VARCHAR(10) UNIQUE NOT NULL,
    train_name VARCHAR(100) NOT NULL,
    train_type ENUM('EXPRESS', 'SUPERFAST', 'PASSENGER', 'RAJDHANI', 'SHATABDI') DEFAULT 'EXPRESS',
    source_station_id INT,
    dest_station_id INT,
    FOREIGN KEY (source_station_id) REFERENCES Stations(station_id),
    FOREIGN KEY (dest_station_id) REFERENCES Stations(station_id)
);

-- 5. SEAT CLASSES (Master list of coach types e.g., 1A, 2A, SL)
CREATE TABLE Seat_Classes (
    class_id INT AUTO_INCREMENT PRIMARY KEY,
    class_code VARCHAR(5) UNIQUE NOT NULL, -- e.g., '1A', 'SL', 'CC'
    class_name VARCHAR(50) NOT NULL,
    fare_multiplier DECIMAL(3,2) NOT NULL -- Used by Java to calculate ticket price
);

-- 6. TRAIN_CLASS_CAPACITY (Maps how many seats of each class a specific train has)
CREATE TABLE Train_Class_Capacity (
    capacity_id INT AUTO_INCREMENT PRIMARY KEY,
    train_id INT,
    class_id INT,
    total_seats INT NOT NULL,
    FOREIGN KEY (train_id) REFERENCES Trains(train_id) ON DELETE CASCADE,
    FOREIGN KEY (class_id) REFERENCES Seat_Classes(class_id) ON DELETE CASCADE
);

-- 7. TRAIN_SCHEDULE (The route map and timetable)
CREATE TABLE Train_Schedule (
    schedule_id INT AUTO_INCREMENT PRIMARY KEY,
    train_id INT,
    station_id INT,
    arrival_time TIME,
    departure_time TIME,
    day_count INT DEFAULT 1, -- e.g., Day 1, Day 2 of the journey
    distance_from_source INT NOT NULL, -- Crucial for fare calculation
    stop_sequence INT NOT NULL,
    FOREIGN KEY (train_id) REFERENCES Trains(train_id) ON DELETE CASCADE,
    FOREIGN KEY (station_id) REFERENCES Stations(station_id)
);

-- 8. BOOKINGS / PNR (The main ticket umbrella)
CREATE TABLE Bookings (
    pnr_number VARCHAR(15) PRIMARY KEY, -- PNRs are usually strings/alphanumeric
    user_id INT,
    train_id INT,
    journey_date DATE NOT NULL,
    booking_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    total_fare DECIMAL(10, 2) NOT NULL,
    booking_status ENUM('CONFIRMED', 'CANCELLED', 'PARTIAL_CANCEL') DEFAULT 'CONFIRMED',
    FOREIGN KEY (user_id) REFERENCES Users(user_id),
    FOREIGN KEY (train_id) REFERENCES Trains(train_id)
);

-- 9. PASSENGERS (Individual people attached to a PNR)
CREATE TABLE Passengers (
    passenger_id INT AUTO_INCREMENT PRIMARY KEY,
    pnr_number VARCHAR(15),
    first_name VARCHAR(50) NOT NULL,
    age INT NOT NULL,
    gender ENUM('M', 'F', 'OTHER') NOT NULL,
    class_id INT,
    seat_number VARCHAR(10), -- e.g., 'B1-42'
    status ENUM('CONFIRMED', 'RAC', 'WL', 'CANCELLED') DEFAULT 'CONFIRMED',
    FOREIGN KEY (pnr_number) REFERENCES Bookings(pnr_number) ON DELETE CASCADE,
    FOREIGN KEY (class_id) REFERENCES Seat_Classes(class_id)
);

-- 10. PAYMENTS (Transaction ledger)
CREATE TABLE Payments (
    payment_id INT AUTO_INCREMENT PRIMARY KEY,
    pnr_number VARCHAR(15),
    amount DECIMAL(10, 2) NOT NULL,
    payment_method ENUM('CREDIT_CARD', 'UPI', 'NET_BANKING', 'DEBIT_CARD'),
    transaction_id VARCHAR(100) UNIQUE,
    payment_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    payment_status ENUM('SUCCESS', 'FAILED', 'REFUNDED') DEFAULT 'SUCCESS',
    FOREIGN KEY (pnr_number) REFERENCES Bookings(pnr_number) ON DELETE CASCADE
);
